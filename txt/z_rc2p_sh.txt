	declare @t_accy nvarchar(max)
	declare @t_addr2 nvarchar(max)
	declare @t_addr nvarchar(max)
	declare @t_tel nvarchar(max)
	declare @t_fax nvarchar(max)
	declare @t_bno nvarchar(20)
	declare @t_eno nvarchar(20)

	set @t_accy = '[1]'
	set @t_addr2 = '[2]'
	set @t_addr = '[3]'
	set @t_tel = '[4]'
	set @t_fax= '[5]'
	
	set @t_bno = case when '#non' = [6] then '' else [6] end
	set @t_eno = case when '#non' = [7] then CHAR(255) else [7] end

declare @tmp table (
		gno nvarchar(10),
		noa nvarchar(30),
		noq nvarchar(10),
		tgg nvarchar(50),
		datea nvarchar(20),
		serial nvarchar(30),
		phone nvarchar(30),
		product nvarchar(max),
		mount float,
		unit nvarchar(20),
		price float,
		total float,
		carno nvarchar(30),
		money float,
		tax float,
		sums float
	)
insert @tmp
select '0',a.noa,b.noq,c.comp,a.datea,c.serial,a.tel,b.product,isnull(b.mount,0)as mount,b.unit,b.price,b.total,a.carno,0,isnull(d.tax,0)as tax,0 
from view_rc2 a
left join view_rc2s  b on a.noa=b.noa
left join tgg c on a.tggno=c.noa
left join vcca d on a.invono = d.noa
where (a.noa between @t_bno and @t_eno)

declare @noa nvarchar(MAX)
declare @counts int 
declare @t_line int = 9

declare cursor_table cursor for
select noa,count(*) from @tmp group by noa
open cursor_table
fetch next from cursor_table
into @noa,@counts
while(@@FETCH_STATUS <> -1)
begin		
	while (@counts<@t_line)
	begin
		insert @tmp(gno,noa,noq)
		select '1',@noa,CHAR(255)
		
		set @counts=@counts+1
	end	
	fetch next from cursor_table
	into @noa,@counts
end
close cursor_table
deallocate cursor_table


insert @tmp(gno,noa,carno,mount,unit,money,tax,sums)
select '2',noa,MAX(carno),SUM(mount),MAX(unit),SUM(total),SUM(tax),SUM(total)+SUM(tax) from @tmp  group by noa

select dbo.getComma(total,0) total,dbo.getComma(money,0) money,dbo.getComma(tax,0) tax,dbo.getComma(sums,0) sums,* ,@t_addr2 addr2,@t_addr addr,@t_tel tel,SUBSTRING(@t_fax,0,CHARINDEX('Fax',@t_fax)) facsimile
from @tmp order by noa,gno,noq  ;
